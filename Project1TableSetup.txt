CREATE TABLE EMPLOYEES(
EMP_ID VARCHAR2(10),
FIRSTNAME VARCHAR2(30),
LASTNAME VARCHAR2(30),
USERNAME VARCHAR2(30),
PASSWORD VARCHAR2(30),
POSITION VARCHAR2(30),
CONSTRAINT PK_EMPLOYEES PRIMARY KEY(EMP_ID));

CREATE TABLE REQUESTS(
REQ_ID VARCHAR2(30),
AMMMOUNT NUMBER(30),
REASON VARCHAR2(250),
STATUS VARCHAR2(30),
CREATED_BY VARCHAR2(30),
RESOLVED_BY VARCHAR2(30),
CONSTRAINT FK_CREATED FOREIGN KEY (CREATED_BY) REFERENCES EMPLOYEES(EMP_ID),
CONSTRAINT FK_RESOLVED FOREIGN KEY (RESOLVED_BY) REFERENCES EMPLOYEES (EMP_ID)
);
